HStack(spacing: 16) {
    ForEach(Clan.allCases, id: \.self) { clan in
        let isSelected = clan == tempSelection

        ClanCard(
            clan: clan,
            image: isSelected ? clan.alternateImage : clan.image
        )
        .onTapGesture {
            HUBPhoneManager.instance.allPlayers[0].userClan = clan
            tempSelection = clan
        }
    }
}
